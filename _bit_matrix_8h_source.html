810 void fec::SparseBitMatrix::RowRef::swap(size_t a, size_t b)
811 {
812  if (a > b) {
813  std::swap(a,b);
814  }
815  auto aIt = std::lower_bound(begin(), end(), a);
816  auto bIt = std::lower_bound(begin(), end(), b);
817  if ((aIt != end() && *aIt == a) && (bIt == end() || *bIt != b)) {
818  for (; aIt < end()-1 && b > aIt[1]; aIt++) {
819  aIt[0] = aIt[1];
820  }
821  *aIt = b;
822  }
823  else if ((aIt == end() || *aIt != a) && (bIt != end() && *bIt == b)) {
824  bIt--;
825  for (; bIt >= begin() && a < bIt[0]; bIt--) {
826  bIt[1] = bIt[0];
827  }
828  bIt[1] = a;
829  }

1093 void fec::BitMatrix::Row::move(size_t a, size_t b)
1094 {
1095  if (a == b) {
1096  return;
1097  }
1098 
1099  size_t first = a;
1100  size_t last = b;
1101  if (a > b) {
1102  std::swap(first, last);
1103  }
1104 
1105  BitField<size_t> before = elements_[first/blocSize()] & ((size_t(1) << (first % blocSize())) - 1);
1106  BitField<size_t> after = elements_[(last+1)/blocSize()] & ~((size_t(1) << ((last+1) % blocSize())) - 1);
1107 
1108  size_t value = elements_[a/blocSize()].test(a%blocSize());
1109  if (a < b) {
1110  if ((a+1) % blocSize() == 0) {
1111  elements_[a/blocSize()] = 0;
1112  }
1113  else {
1114  elements_[a/blocSize()] &= ~( (size_t(1) << ((a+1) % blocSize())) - 1 );
1115  }
1116  elements_[(b+1)/blocSize()] &= (size_t(1) << ((b+1) % blocSize())) - 1;
1117 
1118  size_t carry = 0;
1119  for (int64_t j = b/blocSize(); j >= int64_t(a/blocSize()); --j) {
1120  bool tmp = elements_[j].test(0);
1121  elements_[j] >>= 1;
1122  elements_[j] |= carry << (blocSize()-1);
1123  carry = tmp;
1124  }
1125  }
1126  else {
1127  elements_[a/blocSize()] &= ( (size_t(1) << (a % blocSize())) - 1 );
1128  elements_[b/blocSize()] &= ~( (size_t(1) << (b % blocSize())) - 1 );
1129 
1130  size_t carry = 0;
1131  for (size_t j = b/blocSize(); j <= a/blocSize(); ++j) {
1132  bool tmp = elements_[j].test(blocSize()-1);
1133  elements_[j] <<= 1;
1134  elements_[j] |= carry;
1135  carry = tmp;
1136  }
1137  }
1138  elements_[(last+1)/blocSize()] |= after;
1139  elements_[first/blocSize()] |= before;
1140 
1141  elements_[b/blocSize()] |= value << (b%blocSize());
1142 }
