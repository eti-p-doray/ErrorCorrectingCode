382 void fec::Codec::encode(const std::vector<BitField<size_t>,A<BitField<size_t>>>& msg, std::vector<BitField<size_t>,A<BitField<size_t>>>& parity) const
383 {
384  uint64_t blockCount = msg.size() / (msgSize());
385  if (msg.size() != blockCount * msgSize()) {
386  throw std::invalid_argument("Invalid size for message");
387  }
388 
389  parity.resize(blockCount * paritySize(), 0);
390  auto msgIt = msg.begin(); auto parityIt = parity.begin();
391 
392  auto threadGroup = createWorkGroup();
393  auto thread = threadGroup.begin();
394  size_t step = taskSize(blockCount);
395  for (int i = 0; i + step <= blockCount; i += step) {
396  threadGroup.push_back( std::thread(&Codec::encodeBlocks, this, msgIt, parityIt, step) );
397  msgIt += msgSize() * step;
398  parityIt += paritySize() * step;
399 
400  ++thread;
401  }
402  if (msgIt != msg.end()) {
403  encodeBlocks(msgIt, parityIt, blockCount % step);
404  }
405  for (auto & thread : threadGroup) {
406  thread.join();
407  }

460 void fec::Codec::soDecode(Input<A> input, Output<A> output) const
461 {
462  if (!input.hasParity()) {
463  throw std::invalid_argument("Input must contains parity");
464  }
465  if (input.hasMsg()) {
466  throw std::invalid_argument("Input should not contain msg");
467  }
468  size_t blockCount = input.parity().size() / paritySize();
469  if (input.parity().size() != blockCount * paritySize()) {
470  throw std::invalid_argument("Invalid size for parity");
471  }
472  if (input.hasSyst()) {
473  if (input.syst().size() != blockCount * systSize()) {
474  throw std::invalid_argument("Invalid size for msg");
475  }
476  }
477  if (input.hasState()) {
478  if (input.state().size() != blockCount * stateSize()) {
479  throw std::invalid_argument("Invalid size for state");
480  }
481  }
482 
483  if (output.hasParity()) {
484  output.parity().resize(blockCount * paritySize());
485  }
486  if (output.hasSyst()) {
487  output.syst().resize(blockCount * systSize());
488  }
489  if (output.hasState()) {
490  output.state().resize(blockCount * stateSize());
491  }
492  if (output.hasMsg()) {
493  output.msg().resize(blockCount * msgSize());
494  }
495  auto inputIt = input.begin(structure());
496  auto outputIt = output.begin(structure());
497 
498  auto threadGroup = createWorkGroup();
499  auto thread = threadGroup.begin();
500  size_t step = taskSize(blockCount);
501  for (int i = 0; i + step <= blockCount; i += step) {
502  threadGroup.push_back( std::thread(&Codec::soDecodeBlocks, this,
503  inputIt,
504  outputIt, step
505  ) );
506  inputIt += step;
507  outputIt += step;
508 
509  thread++;
510  }
511  if (outputIt != output.end(structure())) {
512  soDecodeBlocks(inputIt, outputIt, blockCount % step);
513  }
514  for (auto & thread : threadGroup) {
515  thread.join();
516  }
