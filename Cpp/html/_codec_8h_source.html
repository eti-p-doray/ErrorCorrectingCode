171 template <template <typename> class A>
172 void fec::Codec::encode(const std::vector<BitField<size_t>,A<BitField<size_t>>>& msg, std::vector<BitField<size_t>,A<BitField<size_t>>>& parity) const
173 {
174  uint64_t blockCount = msg.size() / (msgSize());
175  if (msg.size() != blockCount * msgSize()) {
176  throw std::invalid_argument("Invalid size for message");
177  }
178 
179  parity.resize(blockCount * paritySize(), 0);
180  auto msgIt = msg.begin(); auto parityIt = parity.begin();
181 
182  auto threadGroup = createWorkGroup();
183  auto thread = threadGroup.begin();
184  size_t step = taskSize(blockCount);
185  for (int i = 0; i + step <= blockCount; i += step) {
186  threadGroup.push_back( std::thread(&Codec::encodeBlocks, this, msgIt, parityIt, step) );
187  msgIt += msgSize() * step;
188  parityIt += paritySize() * step;
189 
190  ++thread;
191  }
192  if (msgIt != msg.end()) {
193  encodeBlocks(msgIt, parityIt, blockCount % step);
194  }
195  for (auto & thread : threadGroup) {
196  thread.join();
197  }
198 }

250 void fec::Codec::soDecode(Input<A> input, Output<A> output) const
251 {
252  if (!input.hasParity()) {
253  throw std::invalid_argument("Input must contains parity");
254  }
255  if (input.hasMsg()) {
256  throw std::invalid_argument("Input should not contain msg");
257  }
258  size_t blockCount = input.parity().size() / paritySize();
259  if (input.parity().size() != blockCount * paritySize()) {
260  throw std::invalid_argument("Invalid size for parity");
261  }
262  if (input.hasSyst()) {
263  if (input.syst().size() != blockCount * systSize()) {
264  throw std::invalid_argument("Invalid size for msg");
265  }
266  }
267  if (input.hasState()) {
268  if (input.state().size() != blockCount * stateSize()) {
269  throw std::invalid_argument("Invalid size for state");
270  }
271  }
272 
273  if (output.hasParity()) {
274  output.parity().resize(blockCount * paritySize());
275  }
276  if (output.hasSyst()) {
277  output.syst().resize(blockCount * systSize());
278  }
279  if (output.hasState()) {
280  output.state().resize(blockCount * stateSize());
281  }
282  if (output.hasMsg()) {
283  output.msg().resize(blockCount * msgSize());
284  }
285  auto inputIt = input.begin(structure());
286  auto outputIt = output.begin(structure());
287 
288  auto threadGroup = createWorkGroup();
289  auto thread = threadGroup.begin();
290  size_t step = taskSize(blockCount);
291  for (int i = 0; i + step <= blockCount; i += step) {
292  threadGroup.push_back( std::thread(&Codec::soDecodeBlocks, this,
293  inputIt,
294  outputIt, step
295  ) );
296  inputIt += step;
297  outputIt += step;
298 
299  thread++;
300  }
301  if (outputIt != output.end(structure())) {
302  soDecodeBlocks(inputIt, outputIt, blockCount % step);
303  }
304  for (auto & thread : threadGroup) {
305  thread.join();
306  }
307 }
