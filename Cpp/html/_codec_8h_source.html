393 template <template <typename> class A>
394 void fec::Codec::encode(const std::vector<BitField<size_t>,A<BitField<size_t>>>& msg, std::vector<BitField<size_t>,A<BitField<size_t>>>& parity) const
395 {
396  uint64_t blockCount = msg.size() / (msgSize());
397  if (msg.size() != blockCount * msgSize()) {
398  throw std::invalid_argument("Invalid size for message");
399  }
400 
401  parity.resize(blockCount * paritySize(), 0);
402  auto msgIt = msg.begin(); auto parityIt = parity.begin();
403 
404  auto threadGroup = createWorkGroup();
405  auto thread = threadGroup.begin();
406  size_t step = taskSize(blockCount);
407  for (int i = 0; i + step <= blockCount; i += step) {
408  threadGroup.push_back( std::thread(&Codec::encodeBlocks, this, msgIt, parityIt, step) );
409  msgIt += msgSize() * step;
410  parityIt += paritySize() * step;
411 
412  ++thread;
413  }
414  if (msgIt != msg.end()) {
415  encodeBlocks(msgIt, parityIt, blockCount % step);
416  }
417  for (auto & thread : threadGroup) {
418  thread.join();
419  }
420 }

471 template <template <typename> class A>
472 void fec::Codec::soDecode(Input<A> input, Output<A> output) const
473 {
474  if (!input.hasParity()) {
475  throw std::invalid_argument("Input must contains parity");
476  }
477  if (input.hasMsg()) {
478  throw std::invalid_argument("Input should not contain msg");
479  }
480  size_t blockCount = input.parity().size() / paritySize();
481  if (input.parity().size() != blockCount * paritySize()) {
482  throw std::invalid_argument("Invalid size for parity");
483  }
484  if (input.hasSyst()) {
485  if (input.syst().size() != blockCount * systSize()) {
486  throw std::invalid_argument("Invalid size for msg");
487  }
488  }
489  if (input.hasState()) {
490  if (input.state().size() != blockCount * stateSize()) {
491  throw std::invalid_argument("Invalid size for state");
492  }
493  }
494 
495  if (output.hasParity()) {
496  output.parity().resize(blockCount * paritySize());
497  }
498  if (output.hasSyst()) {
499  output.syst().resize(blockCount * systSize());
500  }
501  if (output.hasState()) {
502  output.state().resize(blockCount * stateSize());
503  }
504  if (output.hasMsg()) {
505  output.msg().resize(blockCount * msgSize());
506  }
507  auto inputIt = input.begin(structure());
508  auto outputIt = output.begin(structure());
509 
510  auto threadGroup = createWorkGroup();
511  auto thread = threadGroup.begin();
512  size_t step = taskSize(blockCount);
513  for (int i = 0; i + step <= blockCount; i += step) {
514  threadGroup.push_back( std::thread(&Codec::soDecodeBlocks, this,
515  inputIt,
516  outputIt, step
517  ) );
518  inputIt += step;
519  outputIt += step;
520 
521  thread++;
522  }
523  if (outputIt != output.end(structure())) {
524  soDecodeBlocks(inputIt, outputIt, blockCount % step);
525  }
526  for (auto & thread : threadGroup) {
527  thread.join();
528  }
529 }
