808 void fec::SparseBitMatrix::RowRef::swap(size_t a, size_t b)
809 {
810  if (a > b) {
811  std::swap(a,b);
812  }
813  auto aIt = std::lower_bound(begin(), end(), a);
814  auto bIt = std::lower_bound(begin(), end(), b);
815  if ((aIt != end() && *aIt == a) && (bIt == end() || *bIt != b)) {
816  for (; aIt < end()-1 && b > aIt[1]; aIt++) {
817  aIt[0] = aIt[1];
818  }
819  *aIt = b;
820  }
821  else if ((aIt == end() || *aIt != a) && (bIt != end() && *bIt == b)) {
822  bIt--;
823  for (; bIt >= begin() && a < bIt[0]; bIt--) {
824  bIt[1] = bIt[0];
825  }
826  bIt[1] = a;
827  }

1091 void fec::BitMatrix::Row::move(size_t a, size_t b)
1092 {
1093  if (a == b) {
1094  return;
1095  }
1096 
1097  size_t first = a;
1098  size_t last = b;
1099  if (a > b) {
1100  std::swap(first, last);
1101  }
1102 
1103  BitField<size_t> before = elements_[first/blocSize()] & ((size_t(1) << (first % blocSize())) - 1);
1104  BitField<size_t> after = elements_[(last+1)/blocSize()] & ~((size_t(1) << ((last+1) % blocSize())) - 1);
1105 
1106  size_t value = elements_[a/blocSize()].test(a%blocSize());
1107  if (a < b) {
1108  if ((a+1) % blocSize() == 0) {
1109  elements_[a/blocSize()] = 0;
1110  }
1111  else {
1112  elements_[a/blocSize()] &= ~( (size_t(1) << ((a+1) % blocSize())) - 1 );
1113  }
1114  elements_[(b+1)/blocSize()] &= (size_t(1) << ((b+1) % blocSize())) - 1;
1115 
1116  size_t carry = 0;
1117  for (int64_t j = b/blocSize(); j >= int64_t(a/blocSize()); --j) {
1118  bool tmp = elements_[j].test(0);
1119  elements_[j] >>= 1;
1120  elements_[j] |= carry << (blocSize()-1);
1121  carry = tmp;
1122  }
1123  }
1124  else {
1125  elements_[a/blocSize()] &= ( (size_t(1) << (a % blocSize())) - 1 );
1126  elements_[b/blocSize()] &= ~( (size_t(1) << (b % blocSize())) - 1 );
1127 
1128  size_t carry = 0;
1129  for (size_t j = b/blocSize(); j <= a/blocSize(); ++j) {
1130  bool tmp = elements_[j].test(blocSize()-1);
1131  elements_[j] <<= 1;
1132  elements_[j] |= carry;
1133  carry = tmp;
1134  }
1135  }
1136  elements_[(last+1)/blocSize()] |= after;
1137  elements_[first/blocSize()] |= before;
1138 
1139  elements_[b/blocSize()] |= value << (b%blocSize());
1140 }
