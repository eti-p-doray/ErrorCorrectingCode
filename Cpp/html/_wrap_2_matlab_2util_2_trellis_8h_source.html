53 inline mxArray* toMxArray(const fec::Trellis& trellis)
54 {
55  const char* fieldnames[] = {"numInputSymbols", "numOutputSymbols", "numStates", "nextStates", "outputs"};
56  mxArray* out = mxCreateStructMatrix(1,1,5, fieldnames);
57 
58  mxSetField(out, 0, fieldnames[0], toMxArray(trellis.inputCount()));
59  mxSetField(out, 0, fieldnames[1], toMxArray(trellis.outputCount()));
60  mxSetField(out, 0, fieldnames[2], toMxArray(trellis.stateCount()));
61 
62  std::vector<fec::BitField<size_t>> nextStates(trellis.inputCount() * trellis.stateCount());
63  std::vector<fec::BitField<size_t>> outputs(trellis.inputCount() * trellis.stateCount());
64  for (size_t i = 0; i < trellis.stateCount(); ++i) {
65  for (fec::BitField<size_t> j = 0; j < trellis.inputCount(); j++) {
66  fec::BitField<size_t> input = 0;
67  for (int k = 0; k < trellis.inputSize(); ++k) {
68  input[k] = j[trellis.inputSize()-k-1];
69  }
70  nextStates[i+input*trellis.stateCount()] = trellis.getNextState(i, j);
71  outputs[i+input*trellis.stateCount()] = 0;
72  for (size_t k = 0; k < trellis.outputSize(); k++) {
73  outputs[i+input*trellis.stateCount()][trellis.outputSize()-k-1] = trellis.getOutput(i, j).test(k);
74  }
75  }
76  }
77 
78  mxSetField(out, 0, fieldnames[3], toMxArray(nextStates));
79  mxSetField(out, 0, fieldnames[4], toMxArray(outputs));
80  return out;
81 }
