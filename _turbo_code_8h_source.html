133 void fec::TurboCode::pack(typename std::vector<T>::const_iterator parityIn, typename std::vector<T>::iterator parityOut)
134 {
135  auto parityOutIt = parityOut;
136  auto msgInIt = parityIn;
137  auto parityInIt = parityIn + codeStructure_.msgSize() + codeStructure_.msgTailSize();
138  for (size_t i = 0; i < codeStructure_.msgSize(); ++i) {
139  *parityOut = *msgInIt;
140  ++msgInIt;
141  ++parityOut;
142  for (size_t j = 0; j < codeStructure_.constituentCount(); ++j) {
143  if (i < codeStructure_.constituent(i).blocSize()) {
144  for (size_t k = 0; k < codeStructure_.constituent(i).trellis().outputSize(); ++k) {
145  *parityOutIt = *parityInIt;
146  ++parityOutIt;
147  }
148  }
149  }
150  }
151  parityInIt = parityIn + codeStructure_.msgSize() + codeStructure_.msgTailSize();
152  for (size_t i = 0; i < codeStructure_.constituentCount(); ++i) {
153  parityInIt += codeStructure_.constituent(i).paritySize() - codeStructure_.constituent(i).tailSize() * codeStructure_.constituent(i).trellis().outputSize();
154  for (size_t j = 0; j < codeStructure_.constituent(i).tailSize(); ++j) {
155  for (size_t k = 0; k < codeStructure_.constituent(i).trellis().inputSize(); ++k) {
156  *parityOutIt = *msgInIt;
157  ++parityOutIt;
158  ++msgInIt;
159  }
160  for (size_t k = 0; k < codeStructure_.constituent(i).trellis().outputSize(); ++k) {
161  *parityOutIt = *parityInIt;
162  ++parityOutIt;
163  ++parityInIt;
164  }
165  }
166  }
