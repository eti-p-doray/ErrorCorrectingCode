167 template <template <typename> class A>
168 void fec::Code::encode(const std::vector<uint8_t,A<uint8_t>>& message, std::vector<uint8_t,A<uint8_t>>& parity) const
169 {
170  uint64_t blocCount = message.size() / (msgSize());
171  if (message.size() != blocCount * msgSize()) {
172  throw std::invalid_argument("Invalid size for message");
173  }
174 
175  parity.resize(blocCount * paritySize());
176  std::fill(parity.begin(), parity.end(), 0);
177 
178  std::vector<uint8_t>::const_iterator messageIt = message.begin();
179  std::vector<uint8_t>::iterator parityIt = parity.begin();
180 
181  std::vector<std::thread> threadGroup;
182  threadGroup.reserve(workGroupSize());
183  auto thread = threadGroup.begin();
184  int n = std::thread::hardware_concurrency();
185  if (n > workGroupSize() || n == 0) {
186  n = workGroupSize();
187  }
188  size_t step = (blocCount+n-1)/n;
189  for (int i = 0; i + step <= blocCount; i += step) {
190  threadGroup.push_back( std::thread(&Code::encodeNBloc, this,
191  messageIt, parityIt, step) );
192  messageIt += msgSize() * step;
193  parityIt += paritySize() * step;
194 
195  thread++;
196  }
197  if (messageIt != message.end()) {
198  encodeNBloc(messageIt, parityIt, blocCount % step);
199  }
200  for (auto & thread : threadGroup) {
201  thread.join();
202  }
203 }
