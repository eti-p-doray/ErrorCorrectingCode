352 template <template <typename> class A>
353 void fec::Codec::encode(const std::vector<BitField<bool>,A<BitField<bool>>>& msg, std::vector<BitField<uint8_t>,A<BitField<uint8_t>>>& parity) const
354 {
355  uint64_t blockCount = msg.size() / (msgSize());
356  if (msg.size() != blockCount * msgSize()) {
357  throw std::invalid_argument("Invalid size for message");
358  }
359 
360  parity.resize(blockCount * paritySize(), 0);
361  auto msgIt = msg.begin(); auto parityIt = parity.begin();
362 
363  auto threadGroup = createWorkGroup();
364  auto thread = threadGroup.begin();
365  size_t step = taskSize(blockCount);
366  for (int i = 0; i + step <= blockCount; i += step) {
367  threadGroup.push_back( std::thread(&Codec::encodeBlocks, this, msgIt, parityIt, step) );
368  msgIt += msgSize() * step;
369  parityIt += paritySize() * step;
370 
371  ++thread;
372  }
373  if (msgIt != msg.end()) {
374  encodeBlocks(msgIt, parityIt, blockCount % step);
375  }
376  for (auto & thread : threadGroup) {
377  thread.join();
378  }
379 }

430 template <template <typename> class A>
431 void fec::Codec::soDecode(Input<A> input, Output<A> output) const
432 {
433  if (!input.hasParity()) {
434  throw std::invalid_argument("Input must contains parity");
435  }
436  if (input.hasMsg()) {
437  throw std::invalid_argument("Input should not contain msg");
438  }
439  size_t blockCount = input.parity().size() / paritySize();
440  if (input.parity().size() != blockCount * paritySize()) {
441  throw std::invalid_argument("Invalid size for parity");
442  }
443  if (input.hasSyst()) {
444  if (input.syst().size() != blockCount * systSize()) {
445  throw std::invalid_argument("Invalid size for msg");
446  }
447  }
448  if (input.hasState()) {
449  if (input.state().size() != blockCount * stateSize()) {
450  throw std::invalid_argument("Invalid size for state");
451  }
452  }
453 
454  if (output.hasParity()) {
455  output.parity().resize(blockCount * paritySize());
456  }
457  if (output.hasSyst()) {
458  output.syst().resize(blockCount * systSize());
459  }
460  if (output.hasState()) {
461  output.state().resize(blockCount * stateSize());
462  }
463  if (output.hasMsg()) {
464  output.msg().resize(blockCount * msgSize());
465  }
466  auto inputIt = input.begin(structure());
467  auto outputIt = output.begin(structure());
468 
469  auto threadGroup = createWorkGroup();
470  auto thread = threadGroup.begin();
471  size_t step = taskSize(blockCount);
472  for (int i = 0; i + step <= blockCount; i += step) {
473  threadGroup.push_back( std::thread(&Codec::soDecodeBlocks, this,
474  inputIt,
475  outputIt, step
476  ) );
477  inputIt += step;
478  outputIt += step;
479 
480  thread++;
481  }
482  if (outputIt != output.end(structure())) {
483  soDecodeBlocks(inputIt, outputIt, blockCount % step);
484  }
485  for (auto & thread : threadGroup) {
486  thread.join();
487  }
488 }
